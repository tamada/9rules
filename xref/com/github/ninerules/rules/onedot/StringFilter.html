3   import java.util.Map.Entry;
4   import java.util.function.Function;

6   class StringFilter implements Function<String, String>{
7       private Entry<String, String> entry;
8   
9       public StringFilter(Entry<String, String> entry){
10          this.entry = entry;
11      }
12  
13      @Override
14      public String apply(String string) {
15          String key = entry.getKey();
16          String value = entry.getValue();
17          return string.replaceAll(key, value);
18      }
19  }
